197 {
198 TemplateBodyNodePtr<BodyNodeT> bodyNode = mWeakBodyNodePtr.lock();
199 if(nullptr == bodyNode)
200 return nullptr;
201
202 std::shared_ptr<NodeDestructor> destructor = mWeakDestructor.lock();

206 return TemplateNodePtr<NodeT, BodyNodeT>(mNode);

212 if(nullptr == _ptr)
213 {
214 mNode = nullptr;

220 mWeakBodyNodePtr = _ptr->getBodyNodePtr();
221 mWeakDestructor = _ptr->getOrCreateDestructor();
222 mNode = _ptr;
