52template <class Base>
53const std::string& EntityNode<Base>::setName(const std::string& newName)
54{
55 using NameImpl = detail::EntityNodeAspectBase<Base>;
56
57 if(NameImpl::mAspectProperties.mName == newName && !newName.empty())
58 return NameImpl::mAspectProperties.mName;
59
60 const std::string oldName = NameImpl::mAspectProperties.mName;
61
62 NameImpl::mAspectProperties.mName = Node::registerNameChange(newName);
63
64 Node::incrementVersion();
65 Entity::mNameChangedSignal.raise(
66 this, oldName, NameImpl::mAspectProperties.mName);
67
68 return NameImpl::mAspectProperties.mName;
69}
