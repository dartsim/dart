48// Ref: https://bitbucket.org/eigen/eigen/commits/f5b7700
49// TODO: Remove this and use Eigen::aligned_allocator once new version of Eigen
50// is released with above commit.
51template <class T>
52class aligned_allocator_cpp11 : public std::allocator<T>
53{
54public:
55 typedef std::size_t size_type;
56 typedef std::ptrdiff_t difference_type;
57 typedef T* pointer;
58 typedef const T* const_pointer;
59 typedef T& reference;
60 typedef const T& const_reference;
61 typedef T value_type;
62
63 template <class U>
64 struct rebind
65 {
66 typedef aligned_allocator_cpp11<U> other;
67 };
68
69 aligned_allocator_cpp11()
70 : std::allocator<T>() {}
71
72 aligned_allocator_cpp11(const aligned_allocator_cpp11& other)
73 : std::allocator<T>(other) {}
74
75 template <class U>
76 aligned_allocator_cpp11(const aligned_allocator_cpp11<U>& other)
77 : std::allocator<T>(other) {}
78
79 ~aligned_allocator_cpp11() {}
80
81 pointer allocate(size_type num, const void* /*hint*/ = 0)
82 {
83 Eigen::internal::check_size_for_overflow<T>(num);
84 return static_cast<pointer>( Eigen::internal::aligned_malloc(num * sizeof(T)) );
85 }
86
87 void deallocate(pointer p, size_type /*num*/)
88 {
89 Eigen::internal::aligned_free(p);
90 }
91};
