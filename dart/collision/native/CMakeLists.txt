foreach(compat_mapping ${collision_native_compat_header_mappings})
  string(REPLACE "|" ";" compat_mapping_parts "${compat_mapping}")
  list(GET compat_mapping_parts 0 compat_header)
  list(GET compat_mapping_parts 1 canonical_header)

  set(
    compat_header_path
    "${CMAKE_CURRENT_BINARY_DIR}/compat_headers/${compat_header}"
  )
  get_filename_component(compat_header_dir "${compat_header_path}" DIRECTORY)
  file(MAKE_DIRECTORY "${compat_header_dir}")
  file(
    WRITE "${compat_header_path}"
    "#pragma once\n\n// Compatibility forwarding header generated by CMake.\n#include <dart/collision/native/${canonical_header}>\n"
  )

  get_filename_component(compat_install_dir "${compat_header}" DIRECTORY)
  if(compat_install_dir STREQUAL "")
    set(compat_destination "${CMAKE_INSTALL_INCLUDEDIR}/dart/collision/native")
  else()
    set(
      compat_destination
      "${CMAKE_INSTALL_INCLUDEDIR}/dart/collision/native/${compat_install_dir}"
    )
  endif()
  get_filename_component(compat_header_name "${compat_header}" NAME)

  install(
    FILES "${compat_header_path}"
    DESTINATION "${compat_destination}"
    RENAME "${compat_header_name}"
    COMPONENT headers
  )
endforeach()

foreach(native_header ${collision_native_headers})
  file(
    RELATIVE_PATH
    native_header_rel
    "${CMAKE_CURRENT_SOURCE_DIR}"
    "${native_header}"
  )
  get_filename_component(native_header_dir "${native_header_rel}" DIRECTORY)

  if(native_header_dir STREQUAL "")
    set(
      native_header_destination
      "${CMAKE_INSTALL_INCLUDEDIR}/dart/collision/native"
    )
  else()
    set(
      native_header_destination
      "${CMAKE_INSTALL_INCLUDEDIR}/dart/collision/native/${native_header_dir}"
    )
  endif()

  install(
    FILES "${native_header}"
    DESTINATION "${native_header_destination}"
    COMPONENT headers
  )
endforeach()
