44template <typename Derived>

46{

48
49 UniformScalarFromMatrixFunctor(
50 const Eigen::MatrixBase<Derived>& min,
51 const Eigen::MatrixBase<Derived>& max)
52 : mMin(min), mMax(max)
53 {
54 // Do nothing
