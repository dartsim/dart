77template <std::size_t N>
78Eigen::Matrix<double, N, 1> toVectorNd(const std::string& str)
79{
80 Eigen::Matrix<double, N, 1> ret = Eigen::Matrix<double, N, 1>::Zero();
81
82 std::vector<std::string> pieces;
83 std::string trimedStr = boost::trim_copy(str);
84 boost::split(
85 pieces, trimedStr, boost::is_any_of(" "), boost::token_compress_on);
86 std::size_t sizeToRead = std::min(N, pieces.size());
87 if (pieces.size() < N)
88 {
89 dterr << "Failed to read a vector because the dimension '" << pieces.size()
90 << "' is less than the expectation '" << N << "'.\n";
91 }
92 else if (pieces.size() > N)
93 {
94 dterr << "Failed to read a vector because the dimension '" << pieces.size()
95 << "' is greater than the expectation '" << N << "'.\n";
96 }
97
98 for (std::size_t i = 0; i < sizeToRead; ++i)
99 {
100 if (pieces[i] != "")
101 {
102 try
103 {
104 ret(i) = boost::lexical_cast<double>(pieces[i].c_str());
105 }
106 catch (boost::bad_lexical_cast& e)
107 {
108 dterr << "value [" << pieces[i]
109 << "] is not a valid double for Eigen::Vector" << N << "d[" << i
110 << "]: " << e.what() << "\n";
111 }
112 }
113 }
114
115 return ret;
116}

218template <typename ElementType>
219class TemplatedElementEnumerator
220{
221protected:
222 using ElementPtr = ElementType*;
223 using ElementRef = ElementType&;
224
225public:
227 TemplatedElementEnumerator(
228 ElementPtr parentElement, const std::string& childElementName)
229 : mParentElement(parentElement),
230 mChildElementName(childElementName),
231 mCurrentElement(nullptr)
232 {
233 }
234
236 ~TemplatedElementEnumerator() {}
237
240 bool next()
241 {
242 if (!mParentElement)
243 return false;
244
245 if (mCurrentElement)
246 {
247 mCurrentElement
248 = mCurrentElement->NextSiblingElement(mChildElementName.c_str());
249 }
250 else
251 {
252 mCurrentElement
253 = mParentElement->FirstChildElement(mChildElementName.c_str());
254 }
255
256 if (!valid())
257 mParentElement = nullptr;
258
259 return valid();
260 }
261
263 ElementPtr get() const
264 {
265 return mCurrentElement;
266 }
267
269 ElementPtr operator->() const
270 {
271 return mCurrentElement;
272 }
273
275 ElementRef operator*() const
276 {
277 return *mCurrentElement;
278 }
279
281 bool operator==(const TemplatedElementEnumerator<ElementType>& rhs) const
282 {
283 // If they point at the same node, then the names must match
284 return (this->mParentElement == rhs.mParentElement)
285 && (this->mCurrentElement == rhs.mCurrentElement)
286 && (this->mCurrentElement != nullptr
287 || (this->mChildElementName == rhs.mChildElementName));
288 }
289
291 TemplatedElementEnumerator<ElementType>& operator=(
292 const TemplatedElementEnumerator<ElementType>& rhs)
293 {
294 this->mParentElement = rhs.mParentElement;
295 this->mChildElementName = rhs.mChildElementName;
296 this->mCurrentElement = rhs.mCurrentElement;
297
298 return *this;
299 }
300
301private:
303 bool valid() const
304 {
305 return mCurrentElement != nullptr;
306 }
307
308private:
310 ElementPtr mParentElement;
311
313 std::string mChildElementName;
314
316 ElementPtr mCurrentElement;
317};
