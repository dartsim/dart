209 TemplateNodePtr<NodeT, BodyNodeT> lock() const
210 {
211 TemplateBodyNodePtr<BodyNodeT> bodyNode = mWeakBodyNodePtr.lock();
212 if (nullptr == bodyNode)
213 return nullptr;
214
215 std::shared_ptr<NodeDestructor> destructor = mWeakDestructor.lock();
216 if (nullptr == destructor)
217 return nullptr;
218
219 return TemplateNodePtr<NodeT, BodyNodeT>(mNode);

224 {
225 if (nullptr == _ptr)
226 {
227 mNode = nullptr;
228 mWeakDestructor.reset();
229 mWeakBodyNodePtr = nullptr;
230 return;
231 }
232
233 mWeakBodyNodePtr = _ptr->getBodyNodePtr();
234 mWeakDestructor = _ptr->getOrCreateDestructor();
235 mNode = _ptr;
236 }
