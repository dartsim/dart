45template <class CompositeType>
46CompositeTrackingAspect<CompositeType>::CompositeTrackingAspect()
47 : mComposite(
48 nullptr) // This will be set later when the Composite calls setComposite
49{
50 // Do nothing
51}
52
53//==============================================================================
54template <class CompositeType>
55CompositeType* CompositeTrackingAspect<CompositeType>::getComposite()
56{
57 return mComposite;
58}
59
60//==============================================================================
61template <class CompositeType>
62const CompositeType* CompositeTrackingAspect<CompositeType>::getComposite()
63 const
64{
65 return mComposite;
66}
67
68//==============================================================================
69template <class CompositeType>
70bool CompositeTrackingAspect<CompositeType>::hasComposite() const
71{
72 return (nullptr != mComposite);
73}
74
75//==============================================================================
76template <class CompositeType>
77void CompositeTrackingAspect<CompositeType>::setComposite(
78 Composite* newComposite)
79{
80 assert(nullptr == mComposite);
81
82 mComposite = dynamic_cast<CompositeType*>(newComposite);
83 // Note: Derived classes should be responsible for handling the case in which
84 // the new composite type does not match the expected type. We should not
85 // assume here that it is an error.
86}
87
88//==============================================================================
89template <class CompositeType>
90void CompositeTrackingAspect<CompositeType>::loseComposite(
91 Composite* oldComposite)
92{
93 DART_UNUSED(oldComposite);
94 assert(oldComposite == mComposite);
95 mComposite = nullptr;
96}
