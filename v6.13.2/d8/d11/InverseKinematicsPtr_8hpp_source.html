297 TemplateInverseKinematicsPtr<InverseKinematicsT, JacobianNodePtrT> lock()
298 const
299 {
300 JacobianNodePtrT jacNode = mWeakJacNode.lock();
301 if (nullptr == jacNode)
302 return nullptr;
303
304 return TemplateInverseKinematicsPtr<InverseKinematicsT, JacobianNodePtrT>(
305 mWeakIK.lock());
306 }
307
309 template <class OtherIkT, class OtherJacNodePtrT>
310 void set(const TemplateInverseKinematicsPtr<OtherIkT, OtherJacNodePtrT>& _ptr)
311 {
312 if (nullptr == _ptr)
313 {
314 mWeakIK = nullptr;
315 mWeakJacNode = nullptr;
316 return;
317 }
318
319 mWeakIK = _ptr.get();
320 mWeakJacNode = _ptr.get()->getEntity();
321 }
322
324 template <class OtherIkT, class OtherJacNodeT>
325 void set(
326 const TemplateWeakInverseKinematicsPtr<OtherIkT, OtherJacNodeT>& _ptr)
327 {
328 mWeakIK = _ptr.mWeakIK;
329 mWeakJacNode = _ptr.mWeakJacNode;
330 }
331
332protected:
334 std::weak_ptr<InverseKinematicsT> mWeakIK;
335
337 JacobianNodePtrT mWeakJacNode;
338};
